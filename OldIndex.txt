      event = stripe.webhooks.constructEvent(
        req.body, // raw body from express.raw()

      );
    } catch (err) {
      console.error("Webhook signature verification failed:", err.message);
      return res.status(400).send("Webhook Error");

    // 1. VENDOR ONBOARDING
    // ---------------------------------------------------------
    if (event.type === "account.created" || event.type === "account.updated") {
      const account = event.data.object;

      const stripeAccountID = account.id;
      const email = account.email;
      const country = account.country;

      if (email) {
        const snap = await db
          .collection("Users")
          .where("Email", "==", email)
          .limit(1)
          .get();

        if (!snap.empty) {
          const userRef = snap.docs[0].ref;

          await userRef.update({
            Role: "Vendor",

            Country: country,
            receiveMassEmails: true,
            payFrequency: "weekly",
            payDay: "monday",
            updatedAt: admin.firestore.FieldValue.serverTimestamp(),
          });

          await stripe.accounts.update(stripeAccountID, {
            settings: {
              payouts: {
                schedule: {
                  interval: "weekly",
                  weekly_anchor: "monday",
                },
              },
            },
          });

          console.log(`Vendor updated: ${email} → weekly payouts`);
        }
      }
    }

    // ---------------------------------------------------------
    // 2. RESERVE SYSTEM — ADD 5% TO RESERVE ON PAYMENT SUCCESS
    // ---------------------------------------------------------
    if (event.type === "payment_intent.succeeded") {
      const pi = event.data.object;
      const vendorId = pi.metadata?.vendorId;
      const reserveAmount = parseInt(pi.metadata?.reserveAmount || "0");
      let country = null;

      if (pi.transfer_data?.destination) {
        const acct = await stripe.accounts.retrieve(
          pi.transfer_data.destination
        );
        country = acct.country;
      }

      if (!vendorId || !reserveAmount) {
        console.warn(
          "Payment succeeded but missing vendorId or reserveAmount"
        );
      } else {
        const vendorRef = db.collection("Users").doc(vendorId);

        await vendorRef.update({
          Country: country,
          reserveBalance: admin.firestore.FieldValue.increment(reserveAmount),

          }),

        );
      }
    }

    // 3. SEND RESPONSE

app.post("/create-order-payment", async (req, res) => {
  const stripe = req.stripe;
  try {
    const { amount, vendorId, customerId, paymentMethodId } = req.body;

    if (!amount || !vendorId || !customerId || !paymentMethodId) {
      return res.status(400).send("Missing required fields");
    }

    // Fetch vendor Stripe account ID
    const vendorSnap = await db.collection("Users").doc(vendorId).get();
    if (!vendorSnap.exists) {
      return res.status(404).send("Vendor not found");
    }

    const { stripeAccountID } = vendorSnap.data();
    if (!stripeAccountID) {
      return res.status(400).send("Vendor missing Stripe account");
    }

    // Calculate 5% reserve
    const reserveAmount = Math.round(amount * 0.05); // amount is in cents

    // Create PaymentIntent
    const paymentIntent = await stripe.paymentIntents.create({
      amount,
      currency: "bzd", // Tropic Pulse charges customers in Belize dollars
      customer: customerId,
      payment_method: paymentMethodId,
      confirm: true,
      application_fee_amount: reserveAmount, // 5% goes to your platform
      transfer_data: {
        destination: stripeAccountID // 95% goes to vendor
      },
      metadata: {
        vendorId,
        reserveAmount
      }
    });

    res.send({
      clientSecret: paymentIntent.client_secret
    });

  } catch (err) {
    console.error("Error creating PaymentIntent:", err);
    res.status(500).send("Error creating payment");
  }
});

// 1️⃣ UNIVERSAL REQUEST PARSER (Automate + CSV + Raw JSON + Query)
// ---------------------------------------------------------
async function parseIncomingRequest(req) {
  console.log("🔵 [parseIncomingRequest] START");

  let payload = {};
  let email = null;
  let emailType = null;
  let logId = null;

  // Helper to safely decode numbers
  const num = (v) => {
    if (v == null) return 0;
    const decoded = decodeURIComponent(String(v));
    if (decoded.includes("|")) {
      return decoded
        .split("|")
        .map(x => Number(x) || 0)
        .reduce((a, b) => a + b, 0);
    }
    const n = Number(decoded);
    return isNaN(n) ? 0 : Number(n.toFixed(2));
  };

  // Garbage detector
  const isGarbage = (v) => {
    if (!v) return true;
    const s = String(v);
    return (
      s.trim() === "" ||
      s.includes("{{") ||
      s.includes("add_more_field") ||
      s.includes("fieldLebal") ||
      s.includes("fieldValue") ||
      s.includes("*")
    );
  };

  // Cleaner with fallback
  const clean = (v, fallback, label = "") => {
    if (isGarbage(v)) {
      console.log(`⚠️ [CLEAN] Replacing garbage field '${label}' → '${fallback}'`);
      return fallback;
    }
    return String(v).trim();
  };

  // 1️⃣ PRIORITY: AppyPie Automate JSON (req.body)
  if (req.method === "POST" && req.body && typeof req.body === "object") {
    console.log("🟢 [BODY] Using req.body JSON");
    payload = req.body;
    console.log("📦 [BODY PAYLOAD]", payload);

    email = payload.email || null;
    emailType = payload.emailType || payload.type || null;
    logId = payload.logId || null;

  } else {
    // 2️⃣ FALLBACK: Raw body (CSV or JSON)
    const rawBody = extractRawBody(req).trim();
    console.log("🟡 [RAW BODY]", rawBody);

    if (!rawBody.startsWith("{") && (rawBody.includes(",") || rawBody.includes("|"))) {
      console.log("🟠 [CSV MODE DETECTED]");
      const parts = rawBody.includes("|")
        ? rawBody.split("|").map(s => s.trim())
        : rawBody.split(",").map(s => s.trim());

      console.log("📤 [CSV PARTS]", parts);

      email = parts[0] || null;
      emailType = parts[1] || null;
      logId = parts[2] || null;

    } else if (rawBody.startsWith("{")) {
      console.log("🟠 [RAW JSON MODE DETECTED]");
      try {
        payload = JSON.parse(rawBody);
        console.log("📦 [RAW JSON PAYLOAD]", payload);
      } catch (err) {
        console.error("❌ [JSON ERROR] Invalid JSON", err);
        throw new Error("Invalid JSON");
      }

      email = payload.email || null;
      emailType = payload.emailType || payload.type || null;
      logId = payload.logId || null;
    }
  }

  // 3️⃣ MERGE QUERY PARAMS INTO PAYLOAD
  console.log("🔵 [QUERY PARAMS]", req.query);
  const merged = { ...payload, ...req.query };
  console.log("🟣 [MERGED PAYLOAD]", merged);

  // 4️⃣ Extract again from merged
  email = merged.email || email;
  emailType = merged.emailType || merged.type || emailType;
  logId = merged.logId || logId;

  // 5️⃣ AUTO‑CLEAN ALL APPYPIE CUSTOM FIELDS
  console.log("🧹 [SANITIZING FIELDS]");

  emailType = clean(emailType, "newUser", "emailType");
  merged.Country = clean(merged.Country, "Belize", "Country");
  merged.SMS = clean(merged.SMS, "No", "SMS");
  merged.role = clean(merged.role, "Customer", "role");
  merged.group = clean(merged.group, "customer", "group");
  merged.receiveMassEmails = clean(merged.receiveMassEmails, "0", "receiveMassEmails");
  merged.pulsepoints = clean(merged.pulsepoints, "0", "pulsepoints");
  merged.function = clean(merged.function, "setup", "function");

  // 6️⃣ Auto-generate logId if missing
  if (!logId) {
    logId = admin.firestore().collection("EmailLogs").doc().id;
    console.log("🆕 [AUTO LOG ID GENERATED]", logId);
  }

  // 7️⃣ Final validation (now safe)
  if (!email) {
    console.error("❌ [ERROR] Missing Email");
    throw new Error("Missing Email");
  }
  if (!emailType) {
    console.log("⚠️ [DEFAULT] Missing emailType → using 'newUser'");
    emailType = "newUser";
  }

  // Normalize
  email = email.toLowerCase();
  emailType = String(emailType).trim().toLowerCase();

  console.log("✅ [FINAL PARSED RESULT]", {
    email,
    emailType,
    logId,
    payload: merged
  });

  console.log("🔵 [parseIncomingRequest] END");

  return { email, emailType, logId, payload: merged, num };
}

async function configurePayoutSettings(stripe, accountId, payFrequency, payDay) {
  try {
    // Fetch Stripe account
    const account = await stripe.accounts.retrieve(accountId);

    // Find Firestore user
    const snap = await admin.firestore()
      .collection("Users")
      .where("stripeAccountID", "==", accountId)
      .limit(1)
      .get();

    if (snap.empty) {
      throw new Error("Missing user");
    }

    const userDoc = snap.docs[0];
    const userRef = userDoc.ref;
    const userData = userDoc.data();

    // Normalize country
    const country = normalizeCountry(account.country ?? userData.Country ?? userData.country);

    // Instant payout rules
    const instantPayoutSupportedCountries = ["US", "GB", "CA", "AU"];

    // Build payout schedule
    const schedule = { interval: payFrequency };
    if (payFrequency === "weekly") {
      schedule.weekly_anchor = payDay;
    }

    const payoutSettings = {
      settings: {
        payouts: {
          schedule,
          ...(instantPayoutSupportedCountries.includes(country)
            ? {}
            : { debit_negative_balances: false })
        }
      }
    };

    // Update Stripe account
    await stripe.accounts.update(accountId, payoutSettings);

    // Update Firestore user
    await userRef.update({
      Country: country,
      payFrequency,
      payDay,
      updatedAt: admin.firestore.FieldValue.serverTimestamp()
    });

    return {
      country,
      instantPayoutsEnabled: instantPayoutSupportedCountries.includes(country)
    };

  } catch (err) {
    console.error("configurePayoutSettings error:", err);
    throw err;
  }
}

function currency(amount, displayCurrency = "$") {
  // 1. Normalize amount into a clean string
  amount = String(amount || "").replace("$", "").replace("BZ$", "");

  // 2. Convert to number safely
  const num = Number(amount);
  const safe = isNaN(num) ? "0.00" : num.toFixed(2);

  // 3. Normalize displayCurrency safely
  displayCurrency = String(displayCurrency || "$").toUpperCase();

  // 4. Convert displayCurrency into correct symbol
  if (displayCurrency === "USD" || displayCurrency === "$") {
    displayCurrency = "$";
  } else {
    displayCurrency = "BZ$";
  }

  // 5. Return formatted result
  return `${displayCurrency}${safe}`;
}

// ------------------------------------------------------

async function determinePayoutCurrency(stripe, stripeAccountID, payoutAmountUSD) {
  // 1. Retrieve connected account
  const account = await stripe.accounts.retrieve(stripeAccountID);
  const accountCurrency = account.default_currency || "usd";

  // 2. Retrieve platform balance
  const platformBalance = await stripe.balance.retrieve();

  // Extract amounts by currency
  const platformAvailable = {};
  platformBalance.available.forEach(b => {
    platformAvailable[b.currency] = b.amount;
  });

  // 3. FX rates (replace with live rates later if needed)
  const FX_RATE_USD_TO_BZD = 2.0;
  const FX_RATE_BZD_TO_USD = 0.5;

  // 4. Decide transfer currency + amount
  let transferCurrency = "usd";       // default fallback
  let transferAmount = payoutAmountUSD; // default amount

  // If connected account is BZD AND platform has BZD funds → send BZD
  if (accountCurrency === "bzd" && platformAvailable["bzd"] > 0) {
    transferCurrency = "bzd";
    transferAmount = payoutAmountUSD * FX_RATE_USD_TO_BZD; // convert USD → BZD
  }

  // 5. Display currency (for emails/UI)
  // Always show the driver's native currency visually
  let displayAmount = payoutAmountUSD;
  let displayCurrency = "$"; // default for USD

  if (accountCurrency === "bzd") {
    displayAmount = payoutAmountUSD * FX_RATE_USD_TO_BZD;
    displayCurrency = "BZ$"; // remove the D automatically
  }

  return {
    accountCurrency,     // "bzd" or "usd"
    platformAvailable,   // { usd: X, bzd: Y }
    transferCurrency,    // "usd" or "bzd"
    transferAmount,      // amount in transfer currency
    displayAmount,       // amount shown to driver
    displayCurrency      // "$" or "BZ$"
  };
}

function clean(str) {
  if (!str) return "—";
  return String(str)
    .replace(/\\/g, "\\\\")   // ESCAPE BACKSLASHES FIRST
    .replace(/\n/g, " ")      // remove newlines
    .replace(/\r/g, " ")      // remove carriage returns
    .replace(/"/g, '\\"')     // escape double quotes
    .replace(/'/g, "\\'")     // escape apostrophes
    .trim();
}

function cleanpay(str) {
  if (!str) return null;

  return String(str)
    .replace(/[\[\]'\"\\]/g, "")  // remove brackets, quotes, and backslashes
    .replace(/\\n|\\r/g, " ")     // remove escaped newlines
    .trim()
    .toLowerCase();
}

function parseMassEmailBoolean(value) {
  if (value === undefined || value === null) return true; // default opt-in

  const v = String(value).toLowerCase().trim();

  if (v === "1" || v === "true") return true;
  if (v === "0" || v === "false") return false;

  return true; // fallback default
}

function parseOrderedBoolean(value) {
  if (value === undefined || value === null) return false;

  const v = String(value).toLowerCase().trim();

  if (v === "1" || v === "true") return true;
  if (v === "0" || v === "false") return false;

  return false; // fallback default
}

async function handleUserGET(req) {
  const q = data;
  const type = q.type;
  const ppfunction = req.body.function || q.function;
  
  if (!type || type.toLowerCase() !== "users") return;

  const email = q.email;
  let receiveMassEmails = parseMassEmailBoolean(req.body.receiveMassEmails ?? q.receiveMassEmails);

  // -----------------------------
  // LOOK UP USER BY EMAIL
  // -----------------------------
  const usersRef = admin.firestore().collection("Users");
  const snapshot = await usersRef.where("Email", "==", email).limit(1).get();

  let userId = snapshot.empty ? usersRef.doc().id : snapshot.docs[0].id;
  const userRef = usersRef.doc(userId);
  const existing = await userRef.get();
  const data = existing.exists ? existing.data() : {};
  // Extract custom fields safely
  const more = q.add_more_field || q["add_more_field"] || {};

  // COUNTRY
  const rawCountry =
    more["fieldLebal:Country*"]?.fieldValue ||
    q.country ||
    q.Country ||
    data.Country ||
    "";

  const country = normalizeCountry(rawCountry);

  // SMS OPT-IN
  const rawSMS =
    more["fieldLebal:SMS Opt-In (Allow Texts)*"]?.fieldValue ||
    q.SMS ||
    data.SMS ||
    "";

  const SMS = rawSMS === "I agree to receive SMS!";
  // -----------------------------
  // PULSEPOINTS LOGIC (Grant / Redeem / Setup / No-send)
  // -----------------------------
  let currentPoints = Number(data.pulsepoints) || 0;
  const sentPulse = q.pulsepoints !== undefined && q.pulsepoints !== "";
  let pulseAmount = sentPulse ? Number(q.pulsepoints) || 0 : undefined;
  let finalPulsePoints = currentPoints;

  if (sentPulse) {
    if (ppfunction === "Grant") {
      finalPulsePoints = currentPoints + pulseAmount;
    } else if (ppfunction === "Redeem") {
      finalPulsePoints = currentPoints - pulseAmount;
    } else if (ppfunction === "setup") {
      finalPulsePoints = pulseAmount;
    }
  } else {
    finalPulsePoints = currentPoints || 0;
  }

  // -----------------------------
  // ROLE LOGIC
  // -----------------------------
  const currentRole = data.Role || null;
  const newRole = (q.role && q.role.trim() !== "") ? q.role : currentRole;
  
  // -----------------------------
  // NAME LOGIC
  // -----------------------------
  const name = data.name || data.Name || "New User";

  // -----------------------------
  // BUILD UPDATE OBJECT
  // -----------------------------
  const updatedUserData = {
    Email: email,
    Name: name,
    Country: country,
    receiveSMS: SMS,
    receiveMassEmails,
    emailPending: false,
    Role: newRole
  };

  // -----------------------------
  // lastEmailSentAt PROTECTION
  // -----------------------------
  if (data.lastEmailSentAt) {
    // Keep existing timestamp
    updatedUserData.lastEmailSentAt = data.lastEmailSentAt;
  } else {
    // Only set if none exists
    updatedUserData.lastEmailSentAt = null;
  }

  // -----------------------------
  // PULSEPOINTS SAVE RULES
  // -----------------------------
  if (sentPulse || !data.pulsepoints) {
    updatedUserData.pulsepoints = finalPulsePoints;
  }

  // -----------------------------
  // TIMESTAMPS
  // -----------------------------
  if (!existing.exists) {
    updatedUserData.createdAt = admin.firestore.FieldValue.serverTimestamp();
    console.log("Creating NEW User:", userId);
  } else {
    updatedUserData.updatedAt = admin.firestore.FieldValue.serverTimestamp();
    console.log("Updating EXISTING User:", userId);
  }

  // -----------------------------
  // SAVE
  // -----------------------------
  await userRef.set(updatedUserData, { merge: true });
}

async function handleOrderGET(req) {
  const q = data;
  const type = q.type;
  const paymentmethod = q.method;
  // Safe number helpers
  const num = (v) => {
    if (v == null) return 0;
    // Decode URL encoding
    const decoded = decodeURIComponent(String(v));
    if (decoded.includes("|")) {
      return decoded
        .split("|")
        .map(x => Number(x) || 0)
        .reduce((a, b) => a + b, 0);
    }
    const n = Number(decoded);
    return isNaN(n) ? 0 : Number(n.toFixed(2));
  };
  const coupon      = num(q.ordercoupon);   // positive
  const discount    = num(q.orderdiscount); // positive
  const itemcount = num(q.productquantity);
  const itemprices = num(q.orderprice);
  const itemName = q.itemName;
  if (type.toLowerCase() !== "order") return;
  
  // Use provided orderId OR generate one
  const orderID = q.orderID || admin.firestore().collection("Orders").doc().id;
  // Check if order already exists
  const orderRef = admin.firestore().collection("Orders").doc(orderID);
  const existing = await orderRef.get();
  const ordered = parseOrderedBoolean(q.ordered);
  const tip = Number(q.ordertip) || 0;
  const tax = Number(q.ordertax) || 0;
  const shipping = Number(q.ordershipping) || 0;
  const price = itemprices;
  const count = itemcount;
  const payout = Number((tip + (price * 0.05)).toFixed(2));
  const total       = num(q.ordertotal);
  const orderstatus = q.orderstatus;
  
  // What customer actually paid
  // What Tropic Pulse would have earned with NO discounts/coupons
  const pulseIfNoDiscounts =
    (total + coupon + discount) - (price + tax + payout + shipping);

  // What Tropic Pulse actually earned after discounts/coupons
  const pulseActual =
    total - (price + tax + payout + shipping);

  const orderData = {
    orderID,
    ordered,
    paymentmethod,
    estimatedPulseFunds: num(pulseIfNoDiscounts),
    actualPulseFunds: num(pulseActual),
    orderdiscount: num(discount) || 0,
    ordercoupon: num(coupon) || 0,
    customerEmail: q.customerEmail?.toLowerCase() || null,
    delivererEmail: q.delivererEmail?.toLowerCase() || null,
    itemName: itemName || null,
    payoutAmount: payout || null,
    displayAmount: q.displayAmount || payout || 0,
    displayCurrency: q.displayCurrency || "usd",
    transferCurrency: q.transferCurrency || "usd",
    orderprice: price,
    ordertip: num(payout) || null,
    ordertax: num(tax),
    ordershipping: num(shipping),
    ordertotal: num(total),
    orderstatus,
    productquantity: count,
  };

  if (orderstatus === "Pending") {
    orderData.orderedAt =  admin.firestore.FieldValue.serverTimestamp();
  } else if (orderstatus === "Complete") {
    orderData.deliveredAt = admin.firestore.FieldValue.serverTimestamp();
  }

  // If new order → add createdAt
  if (!existing.exists) {
    orderData.createdAt = admin.firestore.FieldValue.serverTimestamp();
    console.log("Creating NEW order:", orderID);
  } else {
    orderData.updatedAt = admin.firestore.FieldValue.serverTimestamp();
    console.log("Updating EXISTING order:", orderID);
  }

  await orderRef.set(orderData, { merge: true });
}

async function checkOrCreateStripeAccount(email, country) {
  const stripe = new Stripe(STRIPE_PASSWORD.value());

  // Normalize email
  const cleanEmail = email.trim().toLowerCase();
  // Normalize country
  const thecountry = normalizeCountry(country);
  // 1. Lookup user by email
  const snap = await admin.firestore()
    .collection("Users")
    .where("Email", "==", cleanEmail)
    .limit(1)
    .get();

  if (snap.empty) {
    throw new Error(`User not found for email: ${cleanEmail}`);
  }

  const userDoc = snap.docs[0];
  const userRef = userDoc.ref;
  const userData = userDoc.data();

  const role = userData.Role;
  const existingStripeID = userData.stripeAccountID;

  // ⭐ PRIORITY: Use payFrequency from Firestore if it exists
  let payFrequency = userData.payFrequency || null;

  // If missing, fall back to role defaults
  if (!payFrequency) {
    if (role === "Deliverer") payFrequency = "daily";
    if (role === "Vendor") payFrequency = "weekly";
  }

  // ⭐ Determine payDay (priority: user record → fallback)
  let payDay = null;

  if (payFrequency === "weekly") {
    // Use stored payDay if exists
    if (userData.payDay) {
      payDay = userData.payDay.toLowerCase();
    } else {
      // Default weekly anchor
      payDay = "monday";
    }
  }

  // Build Stripe payout schedule based on payFrequency + payDay
  let schedule = {};

  if (payFrequency === "daily") {
    schedule = { interval: "daily" };
  }

  if (payFrequency === "weekly") {
    schedule = {
      interval: "weekly",
      weekly_anchor: payDay
    };
  }

  // ⭐ 2. If user already has a Stripe account, update payout schedule
  if (existingStripeID) {
    const account = await stripe.accounts.update(existingStripeID, {
      settings: {
        payouts: { schedule }
      }
    });

    return {
      stripeAccountID: account.id,
      thecountry,
      role,
      payFrequency,
      payDay
    };
  }

  // ⭐ 3. Otherwise create a ne w account
  let stripeAccountID = null;

  try {
    const account = await stripe.accounts.create({
      type: "express",
      country: thecountry,
      email: cleanEmail,
      capabilities: {
        transfers: { requested: true }
      },
      settings: {
        payouts: { schedule }
      }
    });

    stripeAccountID = account.id;
    await configurePayoutSettings(stripe, stripeAccountID, payFrequency, payDay);
    console.log("Created ne w Stripe account:", stripeAccountID);

  } catch (err) {
    console.error("Stripe account creation error:", err);

    // 4. Fallback: search Stripe by email
    const search = await stripe.accounts.search({
      query: `email:'${cleanEmail}'`
    });

    if (!search.data.length) {
      throw new Error(`Stripe account exists but cannot be retrieved for ${cleanEmail}`);
    }

    stripeAccountID = search.data[0].id;
    console.log("Found existing Stripe account:", stripeAccountID);
  }

  // ⭐ 5. Save Stripe ID + payFrequency + payDay to Firestore
  await userRef.set(
    {
      stripeAccountID,
      Country: thecountry,
      payFrequency,
      payDay,
      updatedAt: admin.firestore.FieldValue.serverTimestamp()
    },
    { merge: true }
  );

  // ⭐ 6. Return all values
  return {
    stripeAccountID,
    thecountry,
    role,
    payFrequency,
    payDay
  };
}

/* ===========================
   PAY FREQUENCY/ROLE CHANGE
=========================== */
export const rolechange = onRequest(
  {
    region: "us-central1",
    secrets: [EMAIL_PASSWORD, STRIPE_PASSWORD, SMTP_USER]
  },
  async (req, res) => {
    try {
      const stripe = new Stripe(STRIPE_PASSWORD.value());
      const email = req.query.email;
      const role = req.query.role;
      let payFrequency = cleanpay(req.query.payFrequency);
      let payDay = null;
      let account = null;
      
  // ⭐ Determine payDay (priority: user record → fallback)
 
      if (payFrequency === "weekly") {
        const rawPayDay = cleanpay(req.query.payDay);

        if (!rawPayDay) {
          payDay = "monday";
        } else if (rawPayDay.includes("everyday")) {
          payDay = null;
        } else {
          payDay = rawPayDay;
        }
      }
      if (!payFrequency) {
        if (role === "Deliverer") payFrequency = "daily";
        if (role === "Vendor") payFrequency = "weekly";
      }


      // Build Stripe payout schedule based on payFrequency + payDay
      let schedule = {};

      if (payFrequency === "daily") {
        schedule = { interval: "daily" };
      }

      if (payFrequency === "weekly") {
        schedule = {
          interval: "weekly",
          weekly_anchor: payDay
        };
      }
      let snap = null;
      
      if (email) {
        snap = await admin.firestore()
        .collection("Users")
        .where("Email", "==", email)
        .limit(1)
        .get();
      }
      if (snap.empty) {
        return res.status(400).json({ error: "Missing user" });
      }

      const userDoc = snap.docs[0];
      const userRef = userDoc.ref;
      const userData = userDoc.data();
      const name = userData.Name;
      const uid = userDoc.id;
      const existingStripeID = userData.stripeAccountID;
      const paymentsetup = userData.PaymentSetup;

      account = await stripe.accounts.retrieve(existingStripeID);
      // Normalize country
      const country = normalizeCountry(account.country ?? userData.Country ?? userData.country);

      if (paymentsetup !== "Complete") {
        return res.status(400).json({ error: "Please Finish your Payment Setup First!" });
      }

      if (!uid) {
        return res.status(400).json({ error: "Missing user ID" });
      }
      
      // ⭐ 2. If user already has a Stripe account, update payout schedule
      if (existingStripeID) {
        account = await stripe.accounts.update(existingStripeID, {
          settings: {
            payouts: { schedule }
          }
        });
      }

      await userRef.update({
        Role: role,
        Country: country,
        payFrequency,
        payDay,
        updatedAt: admin.firestore.FieldValue.serverTimestamp()
      });

      const payload = {
        name,
        email,
        userID: uid,
        role,
        country,
        stripeAccountID: existingStripeID,
        payFrequency,
        payDay,
        logId: null
      };
      //console.log("LOGID BEFORE CREATION:", payload.logId);

      /* -----------------------------
         5. Create logId if missing
      ----------------------------- */
      if (!payload.logId) {
        const logref = await admin.firestore().collection("EmailLogs").add({
          date: admin.firestore.FieldValue.serverTimestamp(),
          to: email,
          type: "rolechange",
          payload,
          html: "",
          subject: "",
          triggeredBy: "System",
          triggerSource: "rolechange",
          status: "Pending",
          adminUser: "Automate",
          emailType: "rolechange",
          createdAt: admin.firestore.FieldValue.serverTimestamp(),
          timestamp: admin.firestore.FieldValue.serverTimestamp(),
        });

        payload.logId = logref.id;
      }
      /* -----------------------------
         6. Use emailTemplates system
      ----------------------------- */
      const template = emailTemplates["rolechange"];

      if (!template) {
        return res.status(500).send("missing_template_rolechange");
      }

      const subject = template.subject(payload);
      const html = template.html(payload);
      const finalHeaders = template.headers || (template.important
        ? {
            "X-Priority": "1",
            "X-MSMail-Priority": "High",
            "Importance": "high"
          }
        : {}
      );

      /* -----------------------------
         7. Update log with HTML + subject
      ----------------------------- */
      
      await admin.firestore().collection("EmailLogs").doc(payload.logId).set(
        {
          payload,
          html,
          subject,
          updatedAt: admin.firestore.FieldValue.serverTimestamp()
        },
        { merge: true }
      );

      /* -----------------------------
         8. Send email
      ----------------------------- */
      const transporter = nodemailer.createTransport({
        service: "gmail",
        port: 465,
        secure: true,
        auth: {
          user: "Sales@TropicPulse.bz",
          pass: EMAIL_PASSWORD.value()
        }
      });

      await transporter.sendMail({
        from: `"Tropic Pulse" <Sales@TropicPulse.bz>`,
        to: email,
        bcc: "FordFamilyDelivery@gmail.com",
        subject,
        html,
        headers: finalHeaders
      });

      /* -----------------------------
         9. Mark log as sent
      ----------------------------- */
      await admin.firestore().collection("EmailLogs").doc(payload.logId).update({
        status: "Sent",
        sentAt: admin.firestore.FieldValue.serverTimestamp()
      });



      return res.status(200).json({ status: "success" });

    } catch (err) {
      console.error("ROLECHANGE ERROR:", {
        message: err.message,
        stack: err.stack,
        email: req.query.email,
        role: req.query.role,
        payFrequency: req.query.payFrequency,
        payDay: req.query.payDay
      });

      return res.status(500).json({ error: "server_error" });
    }
  }
);

export const retryMassEmails = onSchedule(
  {
    schedule: "every day 01:00",
    region: "us-central1",
    timeZone: "America/Belize",
    secrets: ["SMTP_USER", "EMAIL_PASSWORD", "STRIPE_PASSWORD"]
  },
  async () => {
    const snap = await admin.firestore()
      .collection("Users")
      .where("emailPending", "==", true)
      .get();

    if (snap.empty) {
      console.log("No pending emails to retry.");
      return;
    }

    const promises = [];

    snap.forEach((doc) => {
      const user = doc.data();
      const userEmail = user.Email;

      // If no email or no stored emailType, skip
      if (!userEmail || !user.pendingEmailType) return;

      // Build payload from stored pending data
      const payload = user.pendingPayload || {};

      // Let sendEmailToUser() handle:
      // - cooldown
      // - opt-out
      // - lastEmailSentAt update
      // - clearing emailPending
      promises.push(
        sendEmailToUser(userEmail, user.pendingEmailType, payload)
      );
    });

    await Promise.all(promises);

    console.log("Retry cycle complete.");
  }
);

export const sendMASSemail = onRequest(
  {
    region: "us-central1",
    secrets: [EMAIL_PASSWORD, STRIPE_PASSWORD, SMTP_USER],
    timeoutSeconds: 540,
    memory: "1GiB"
  },
  async (req, res) => {
    try {
      const q = req.query;
      const emailType = req.body.emailType || q.emailType;
      const payload = req.body.payload || {};
      const eventID = req.body.eventID || q.eventID;
      const busID  = req.body.busID || q.busID;
      
      if (emailType === "newEvent") {
        if (!eventID) {
          console.log("No eventID provided.");
          return res.status(400).json({ error: "Missing eventID" });
        }
      

      // Pull event from Firestore
      const eventDoc = await admin
        .firestore()
        .collection("Events")
        .doc(eventID)
        .get();

      if (!eventDoc.exists) {
        console.log("Event not found:", eventID);
        return res.status(404).json({ error: "Event not found" });
      }

      const eventData = eventDoc.data();

      // Extract the fields you need
      const title = eventData.Title || eventData.title || "";
      const description = eventData.Description || eventData.description || "";
      const summary = eventData.Summary || eventData.summary || "";
      
      payload.title = title;
      payload.description = description;
      payload.summary = summary;
      
      let date = eventData.Date || eventData.date;

      if (date instanceof admin.firestore.Timestamp) {
        payload.date = date.toDate().toISOString();
      } else if (date instanceof Date) {
        payload.date = date.toISOString();
      } else if (typeof date === "string") {
        payload.date = new Date(date).toISOString();
      } else {
        // fallback to now
        payload.date = new Date().toISOString();
      }

      payload.eventID = eventID;
      console.log("Loaded event:", { title, description, summary, date });
    }
    if (emailType === "newBusiness") {
        if (!busID) {
          console.log("No busID provided.");
          return res.status(400).json({ error: "Missing busID" });
        }
      

      // Pull event from Firestore
      const busDoc = await admin
        .firestore()
        .collection("Businesses")
        .doc(busID)
        .get();

      if (!busDoc.exists) {
        console.log("Business not found:", busID);
        return res.status(404).json({ error: "Business not found" });
      }

      const busData = busDoc.data();

      // Extract the fields you need
      const busname = busData.Busname || busData.busname || "";
      const summary = busData.Summary || busData.summary || "";
      const description = busData.Description || busData.description || "";
      const location = busData.Location || busData.location || "";
      
      const busemail = busData.Busemail || busData.busemail || "";
      payload.busname = busname;
      payload.summary = summary;
      payload.description = description;
      payload.location = location;

      let date = busData.Date || busData.date;

      if (date instanceof admin.firestore.Timestamp) {
        payload.date = date.toDate().toISOString();
      } else if (date instanceof Date) {
        payload.date = date.toISOString();
      } else if (typeof date === "string") {
        payload.date = new Date(date).toISOString();
      } else {
        // fallback to now
        payload.date = new Date().toISOString();
      }

      payload.busemail = busemail;
      payload.busID = busID;
      console.log("Loaded business:", { busname, summary, description, location, date, busemail });
    }
      // Only allow the two supported types
      const allowedTypes = ["newEvent", "newBusiness"];

      if (!allowedTypes.includes(emailType)) {
        console.log("Unsupported emailType:", emailType);
        return res.json({
          success: false,
          message: `Email type '${emailType}' is not supported.`,
        });
      }

      console.log("Broadcast email triggered for:", emailType);

      // Pull all users
      const usersSnapshot = await admin
        .firestore()
        .collection("Users")
        .get();

      if (usersSnapshot.empty) {
        return res.json({ success: false, message: "No users found." });
      }

      const sendPromises = [];

      usersSnapshot.forEach((doc) => {
        const user = doc.data();
        const userEmail = user.Email; // IMPORTANT: capital E
        const userID = doc.id;
        payload.userID = userID;
        
        if (!userEmail) return;

        const userPayload = {
          ...payload,
          email: userEmail,
        };

        sendPromises.push(
          sendEmailToUser(userEmail, emailType, userPayload)
        );
      });

      await Promise.all(sendPromises);

      return res.json({
        success: true,
        message: `Broadcast email sent to ${usersSnapshot.size} users.`,
      });

    } catch (err) {
      console.error("broadcastEmail error:", err);
      return res.status(500).json({
        error: "Broadcast email failed",
        details: err.message || String(err),
      });
    }
  }
);

    secrets: [EMAIL_PASSWORD, STRIPE_PASSWORD, SMTP_USER]
  },
  async (req, res) => {
  try {
    const authHeader = req.headers.authorization || "";
    const token = authHeader.startsWith("Bearer ")
      ? authHeader.split("Bearer ")[1]
      : null;

    if (!token) {
      return res.status(403).send("Access Denied: Token Needed");
    }

    const decoded = await admin.auth().verifyIdToken(token);

    // Only YOU can access admin.html
    const allowedEmail = "fordfamilydistribution@gmail.com"; // replace with your Google account

    if (decoded.email !== allowedEmail) {
      return res.status(403).send("Access Denied: Email Needed");
    }

    // Serve admin.html from your hosting folder
    return res.sendFile(
      path.join(__dirname, "../tropic-admin/admin.html")
    );

  } catch (err) {
    console.error("Admin gate error:", err);
    return res.status(403).send("Access Denied: " + err.message);
  }
});

=========================== */
// functions/stripeSetupComplete.js (or inside index.js)


export const stripeSetupComplete = onRequest(
  {
    region: "us-central1",
    secrets: [EMAIL_PASSWORD, STRIPE_PASSWORD, SMTP_USER]
  },
  async (req, res) => {
  try {
    const token = req.query.token;
    const stripe = new Stripe(STRIPE_PASSWORD.value());
    if (!token || typeof token !== "string") {
      console.error("Missing or invalid token");
      return res.redirect("https://stripe.tropicpulse.bz/error.html");
    }

    const snap = await admin.firestore()
      .collection("Users")
      .where("resendToken", "==", token)
      .limit(1)
      .get();

    if (snap.empty) {
      console.error("No user found for token:", token);
      return res.redirect(`https://stripe.tropicpulse.bz/error.html`);
    }

    const userDoc = snap.docs[0];
    const userRef = userDoc.ref;
    const data = snap.docs[0].data();
    const accountId = data.stripeAccountID;
    // -------------------------------
    // LOGIN ATTEMPT COUNTER
    // -------------------------------
    const attempts = (data.loginAttempts || 0) + 1;
    await userRef.update({ loginAttempts: attempts });

    // Throttle Stripe login link creation
    const now = Date.now();
    const lastLogin = data.loginAt || 0;

    // 30 second cooldown
    if (now - lastLogin < 60000 && data.loginLink) {
      return res.redirect(`https://stripe.tropicpulse.bz/SetupComplete.html?to=${data.loginLink}`);
    }
    // Generate a fresh Stripe dashboard login link
    const link = await stripe.accounts.createLoginLink(accountId);

    await userRef.update({
      PaymentSetup: "Complete",
      setupAt: admin.firestore.FieldValue.serverTimestamp(),
      loginAt: now,
      loginLink: link.url
    });

    return res.redirect(`https://stripe.tropicpulse.bz/SetupComplete.html?to=${link.url}`);
  } catch (err) {
    console.error("Error in stripeSetupComplete:", err);
    return res.redirect("https://stripe.tropicpulse.bz/error.html");
  }
});

            "X-MSMail-Priority": "High",
            "Importance": "high"

/* ===========================
   SEND PAYOUT FUNCTION
=========================== */

export const sendPayout = onRequest(
  {
    region: "us-central1",
    secrets: [EMAIL_PASSWORD, STRIPE_PASSWORD, SMTP_USER]
  },
  async (req, res) => {
    let logId = null;
    
    try {
      const stripe = new Stripe(STRIPE_PASSWORD.value());

      // Allow GET and POST
      if (req.method !== "GET" && req.method !== "POST") {
        return res.status(405).send({ error: "Method not allowed" });
      }

      // Extract fields depending on method
      const source = req.method === "GET" ? req.query : req.body;

      const {
        orderID,
        email,
        emailType = "sendPayout"
      } = source;

      const delivererEmail = email || source.delivererEmail || null;
      const adminUser = "Automate";

      // ---------------------------------------------
      // 1. Lookup User by Email → get stripeAccountID
      // ---------------------------------------------
      const userSnap = await admin.firestore()
        .collection("Users")
        .where("Email", "==", delivererEmail)
        .limit(1)
        .get();

      if (userSnap.empty) {
        return res.status(400).send({ error: "User not found for email" });
      }

      const userRef = userSnap.docs[0].ref;
      const userData = userSnap.docs[0].data();
      const stripeAccountID = userData.stripeAccountID;
      const name = userData.Name;

      if (!stripeAccountID) {
        return res.status(400).send({ error: "User has no Stripe account ID" });
      }

      // ---------------------------------------------
      // 2. Lookup Order by orderID → get payoutAmount
      // ---------------------------------------------
      const orderSnap = await admin.firestore()
        .collection("Orders")
        .doc(orderID)
        .get();

      if (!orderSnap.exists) {
        return res.status(400).send({ error: "Order not found" });
      }

      const num = (v) => {
        if (v == null) return 0;
        const decoded = decodeURIComponent(String(v));
        if (decoded.includes("|")) {
          return decoded
            .split("|")
            .map(x => Number(x) || 0)
            .reduce((a, b) => a + b, 0);
        }
        const n = Number(decoded);
        return isNaN(n) ? 0 : Number(n.toFixed(2));
      };

      const orderData = orderSnap.data();
      const payout = num(orderData.tip);
      let payoutAmount = orderData.displayAmount || orderData.payoutAmount || payout || null;
      payoutAmount = num(payoutAmount);

      const itemPrice = num(orderData.orderprice) || null;
      const totalPrice = orderData.ordertotal || null;
      const tipAmount = orderData.ordertip || null;
      const taxAmount = orderData.ordertax || null;
      const shipping = orderData.ordershipping || null;
      const paidDeliverer = orderData.paidDeliverer || false;

      if (!payoutAmount) {
        return res.status(400).send({ error: "Order missing payoutAmount" });
      }
      if (paidDeliverer === true) {
        return res.status(400).send({ error: "Order already paidDeliverer" });
      }

      // ----------------------------------------------
      // 3. Determine currency + balances
      // ----------------------------------------------
      const info = await determinePayoutCurrency(stripe, stripeAccountID, payoutAmount);

      let stripeBalances = await findUserStripeBalance(stripeAccountID);
      let pendingBalance = num(stripeBalances.pendingBalance);
      let availableBalance = num(stripeBalances.availableBalance);

      let displayCurrency = info.displayCurrency;
      let transferCurrency = info.transferCurrency;
      let displayAmount = info.displayAmount;

      // ----------------------------------------------
      // ⭐ NEW: DELIVERER RESERVE LOGIC (5%)
      // ----------------------------------------------
      const delivererReserve = Math.round(info.transferAmount * 0.05);
      const delivererPayout = Math.round(info.transferAmount * 0.95);

      // ----------------------------------------------
      // 4. Stripe Transfer (updated with metadata)
      // ----------------------------------------------
      const transfer = await stripe.transfers.create({
        amount: delivererPayout * 100,
        currency: transferCurrency,
        destination: stripeAccountID,
        description: `Tropic Pulse: Payout for Delivery ${orderID}`,
        metadata: {
          orderID,
          delivererEmail,
          delivererReserve
        }
      });

      // ----------------------------------------------
      // 5. Mark order as paid
      // ----------------------------------------------
      await admin.firestore()
        .collection("Orders")
        .doc(orderID)
        .update({
          paidDeliverer: true,
          paidDelivererID: transfer.id,
          paidDelivererAt: admin.firestore.FieldValue.serverTimestamp()

export const emailOpened = onRequest(async (req, res) => {
  const logId = req.query.logId;

  // Helper: always return pixel
  const pixel = () => sendPixel(res);

  // 1️⃣ Validate logId
  if (!logId || logId === "undefined") {
    console.log("Ignoring invalid logId:", logId);
    return pixel();
  }

  console.log("Pixel fired for:", logId);

  const ref = admin.firestore().collection("EmailLogs").doc(logId);
  const snap = await ref.get();

  // 2️⃣ Ensure log exists
  if (!snap.exists) {
    console.log("Log not found:", logId);
    return pixel();
  }

  const { status } = snap.data();

  // 3️⃣ Skip updates for Failed or Opened
  if (status === "Failed") {
    console.log("Not updating — email FAILED:", logId);
    return pixel();
  }
  if (status === "Pending") {
    console.log("Not updating — email PENDED via Error:", logId);
    return pixel();
  }

  if (status === "Opened") {
    console.log("Already opened — skipping:", logId);
    return pixel();
  }

  // 4️⃣ Update to Opened
  await ref.update({
    openedAt: admin.firestore.FieldValue.serverTimestamp(),
    status: "Opened"
  });

  console.log("Updated Firebase → Opened:", logId);
  return pixel();
});
